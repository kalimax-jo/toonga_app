import 'package:flutter/material.dart';
import 'package:iconsax/iconsax.dart';

import '../../theme/app_colors.dart';

class HomeScreen extends StatefulWidget {
  const HomeScreen({super.key});

  @override
  State<HomeScreen> createState() => _HomeScreenState();
}

class _HomeScreenState extends State<HomeScreen> {
  final PageController _pageController = PageController(viewportFraction: .84);
  final ScrollController _chipController = ScrollController();

  int _selectedCategory = 0;
  int _currentNav = 0;

  final List<_Category> _categories = const [
    _Category('Beverage', 'assets/images/background.png',
        'Wines, spirits & premium drinks'),
    _Category('Nightlife', 'assets/images/yellow_bg.png',
        'Late-night lounges & clubs'),
    _Category('Gourmet', 'assets/images/background.png',
        'Chef curated tasting menus'),
    _Category('Wellness', 'assets/images/yellow_bg.png',
        'Relax & recharge rituals'),
  ];

  @override
  void dispose() {
    _pageController.dispose();
    _chipController.dispose();
    super.dispose();
  }

  @override
  Widget build(BuildContext context) {
    final screenHeight = MediaQuery.of(context).size.height;
    final cardHeight = (screenHeight * 0.5).clamp(320.0, 420.0);
    return Scaffold(
      backgroundColor: AppColors.background,
      body: SafeArea(
        child: SingleChildScrollView(
          padding: const EdgeInsets.only(bottom: 70),
          child: Column(
            crossAxisAlignment: CrossAxisAlignment.start,
            children: [
              Padding(
                padding: const EdgeInsets.fromLTRB(24, 18, 24, 0),
                child: Column(
                  crossAxisAlignment: CrossAxisAlignment.start,
                  children: const [
                    _SocialHeader(),
                    SizedBox(height: 20),
                    Text(
                      'Plan a premium experience',
                      style: TextStyle(
                        color: Colors.white,
                        fontSize: 16,
                        fontWeight: FontWeight.w600,
                        letterSpacing: -0.2,
                      ),
                    ),
                    SizedBox(height: 6),
                    Text(
                      'Select a category that suits your mood today.',
                      style: TextStyle(
                        color: Colors.white60,
                        fontSize: 14,
                      ),
                    ),
                  ],
                ),
              ),
              const SizedBox(height: 24),
              SizedBox(
                height: 46,
                child: ListView.builder(
                  controller: _chipController,
                  scrollDirection: Axis.horizontal,
                  padding: const EdgeInsets.symmetric(horizontal: 20),
                  itemCount: _categories.length,
                  itemBuilder: (context, index) {
                    final selected = index == _selectedCategory;
                    return GestureDetector(
                      onTap: () => _animateTo(index),
                      child: AnimatedContainer(
                        duration: const Duration(milliseconds: 250),
                        curve: Curves.easeOut,
                        margin: EdgeInsets.only(
                          right: index == _categories.length - 1 ? 0 : 12,
                        ),
                        padding: const EdgeInsets.symmetric(
                          horizontal: 18,
                          vertical: 10,
                        ),
                        decoration: BoxDecoration(
                          color: selected
                              ? AppColors.primary
                              : Colors.white.withOpacity(0.06),
                          borderRadius: BorderRadius.circular(30),
                          border: Border.all(
                            color: selected
                                ? AppColors.primary
                                : Colors.white.withOpacity(0.15),
                          ),
                        ),
                        child: Text(
                          _categories[index].title,
                          style: TextStyle(
                            color: selected ? Colors.black : Colors.white70,
                            fontWeight: FontWeight.w600,
                          ),
                        ),
                      ),
                    );
                  },
                ),
              ),
              const SizedBox(height: 24),
              SizedBox(
                height: cardHeight,
                child: PageView.builder(
                  controller: _pageController,
                  padEnds: true,
                  onPageChanged: (value) =>
                      setState(() => _selectedCategory = value),
                  itemCount: _categories.length,
                  itemBuilder: (context, index) {
                    final category = _categories[index];
                    return AnimatedPadding(
                      duration: const Duration(milliseconds: 250),
                      padding: EdgeInsets.only(
                        top: index == _selectedCategory ? 0 : 24,
                        bottom: index == _selectedCategory ? 0 : 24,
                        left: 12,
                        right: 12,
                      ),
                      child: _CategoryCard(category: category),
                    );
                  },
                ),
              ),
              const SizedBox(height: 32),
            ],
          ),
        ),
      ),
      bottomNavigationBar: _BottomNav(
        currentIndex: _currentNav,
        onTap: (index) => setState(() => _currentNav = index),
      ),
    );
  }

  void _animateTo(int index) {
    setState(() => _selectedCategory = index);
    _pageController.animateToPage(
      index,
      duration: const Duration(milliseconds: 350),
      curve: Curves.easeOutCubic,
    );
  }
}

class _BottomNav extends StatelessWidget {
  final int currentIndex;
  final ValueChanged<int> onTap;

  const _BottomNav({required this.currentIndex, required this.onTap});

  @override
  Widget build(BuildContext context) {
    final items = [
      Icons.home_rounded,
      Icons.shopping_bag_outlined,
      Icons.play_circle_fill,
      Icons.account_balance_wallet_outlined,
      Icons.person_outline,
    ];
    final labels = ['Home', 'Cart', 'Reels', 'Pay', 'Profile'];

    return Container(
      margin: const EdgeInsets.fromLTRB(24, 0, 24, 24),
      padding: const EdgeInsets.symmetric(horizontal: 28, vertical: 14),
      decoration: BoxDecoration(
        color: AppColors.primary,
        borderRadius: BorderRadius.circular(36),
        border: Border.all(color: Colors.black.withOpacity(0.05)),
        boxShadow: [
          BoxShadow(
            color: Colors.black.withOpacity(0.25),
            blurRadius: 20,
            offset: const Offset(0, 12),
          ),
        ],
      ),
      child: Row(
        mainAxisAlignment: MainAxisAlignment.spaceBetween,
        children: List.generate(items.length, (index) {
          final selected = currentIndex == index;

          return GestureDetector(
            onTap: () => onTap(index),
            child: Column(
              mainAxisSize: MainAxisSize.min,
              children: [
                Icon(
                  items[index],
                  color: selected ? Colors.white : Colors.black87,
                ),
                const SizedBox(height: 4),
                Text(
                  labels[index],
                  style: TextStyle(
                    color: selected ? Colors.white : Colors.black87,
                    fontWeight: selected ? FontWeight.w700 : FontWeight.w500,
                    fontSize: 11,
                  ),
                ),
              ],
            ),
          );
        }),
      ),
    );
  }
}

class _Category {
  final String title;
  final String image;
  final String subtitle;

  const _Category(this.title, this.image, this.subtitle);
}
